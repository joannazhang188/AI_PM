  const saved = JSON.parse(localStorage.getItem(STORAGE_KEY) || "null");
  return {
    sidebarCollapsed: saved?.sidebarCollapsed || false,

function bindActionNodes(root) {
  root.querySelectorAll("[data-action]").forEach((node) => {
    node.onclick = handleAction;
  });
}

function handleAction(event) {
  const action = event.currentTarget.dataset.action;
  if (action === "toggle-sidebar") {
    state.sidebarCollapsed = !state.sidebarCollapsed;
    render();
    return;
  }
  if (action === "toggle-project") {
    const project = state.projects.find((item) => item.id === event.currentTarget.dataset.projectId);
    if (project) project.expanded = !project.expanded;
    render();
    return;
  }
  if (action === "new-project" || action === "new-conversation") {
    goHome();
    return;
  }
  if (action === "open-library" || action === "open-library-from-composer") {
    state.libraryReturnToConversation = state.mainView === "conversation";
    state.mainView = "library";
    render();
    return;
  }
  if (action === "back-from-library") {
    state.mainView = state.libraryReturnToConversation && state.lastConversationId ? "conversation" : "home";
    state.activeConversationId = state.libraryReturnToConversation ? state.lastConversationId : "";
    state.documentOpen = false;
    render();
    return;
  }
  if (action === "shortcut") {
    const shortcut = event.currentTarget.dataset.shortcut;
    startConversation("conv-2", "project-2", shortcut === "需求评审" ? "ready" : "blank", shortcut);
    return;
  }
  if (action === "open-conversation") {
    const { projectId, conversationId } = event.currentTarget.dataset;
    const selected = findConversation(projectId, conversationId);
    startConversation(conversationId, projectId, selected?.kind || "blank");
    return;
  }
  if (action === "send-message") {
    startConversation("conv-2", "project-2", "ready", state.composerMode || "需求评审");
    return;
  }
  if (action === "exit-conversation") {
    goHome();
    return;
  }
  if (action === "open-document") {
    state.documentOpen = true;
    state.activeDocumentId = event.currentTarget.dataset.documentId;
    render();
    return;
  }
  if (action === "close-document") {
    state.documentOpen = false;
    render();
    return;
  }
  if (action === "switch-doc-mode") {
    state.docMode = event.currentTarget.dataset.mode;
    render();
    return;
  }
  if (action === "quote-resource") {
    state.mainView = "conversation";
    state.activeConversationId = state.lastConversationId || "conv-2";
    render();
  }
}

function goHome() {
  state.mainView = "home";
  state.activeConversationId = "";
  state.activeProjectId = "";
  state.documentOpen = false;
  state.composerMode = "";
  render();
}

function startConversation(conversationId, projectId, templateKey = "blank", shortcutName = "") {
  state.mainView = "conversation";
  state.activeConversationId = conversationId;
  state.activeProjectId = projectId;
  state.lastConversationId = conversationId;
  state.documentOpen = false;
  if (shortcutName) state.composerMode = shortcutName;
  const project = state.projects.find((item) => item.id === projectId);
  const exists = project?.conversations.find((item) => item.id === conversationId);
  if (!exists && project) {
    project.conversations.unshift({ id: conversationId, title: shortcutName || "新对话", time: "刚刚", kind: templateKey });
  } else if (exists) {
    exists.kind = templateKey;
  }
  render();
}

function getActiveConversation() {
  if (!state.activeConversationId) return null;
  const selected = findConversation(state.activeProjectId, state.activeConversationId);
  return getTemplateConversation(selected?.kind || "blank");
}

  return project?.conversations.find((item) => item.id === conversationId) || null;
}

    .replaceAll("'", "&#39;");
}
